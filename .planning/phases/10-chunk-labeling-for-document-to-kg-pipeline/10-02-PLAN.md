import pytest
import pytest_asyncio
from unittest.mock import AsyncMock, patch, MagicMock
from api.kg_processor import (
    Chunk,
    KnowledgeGraphProcessor,
    GeminiClient,

@pytest.fixture
def processor():

class TestExtractJsonArray:
    """Tests for _extract_json_array method."""

    def test_valid_json_array(self, processor):
        result = processor._extract_json_array(
            '[["Label A", "Label B"], ["Label C"]]'
        )
        assert result == [["Label A", "Label B"], ["Label C"]]

    def test_json_array_with_surrounding_text(
        self, processor
    ):
        result = processor._extract_json_array(
            'Here are labels: [["A"], ["B"]] done.'
        )
        assert result == [["A"], ["B"]]

    def test_no_brackets(self, processor):
        result = processor._extract_json_array(
            "no json here"
        )
        assert result == []

    def test_malformed_json(self, processor):
        result = processor._extract_json_array(
            '[["unclosed'
        )
        assert result == []

    def test_empty_array(self, processor):
        result = processor._extract_json_array("[]")
        assert result == []


class TestHeuristicLabel:
    """Tests for _heuristic_label method."""

    def test_short_first_sentence(self, processor):
        result = processor._heuristic_label(
            "Thermodynamics basics. More text here."
        )
        assert result == ["Thermodynamics basics"]

    def test_long_first_sentence_truncates(
        self, processor
    ):
        long_text = "A" * 100 + ". Second sentence."
        result = processor._heuristic_label(long_text)
        assert len(result) == 1
        assert len(result[0]) == 63  # 60 + "..."
        assert result[0].endswith("...")

    def test_no_period(self, processor):
        result = processor._heuristic_label(
            "Short text without period"
        )
        assert result == ["Short text without period"]


@pytest.mark.asyncio
class TestGenerateChunkLabels:
    """Tests for _generate_chunk_labels method."""

    async def test_success(
        self, processor, sample_chunks
    ):
        """LLM returns correct number of label lists."""
        processor.gemini.generate_text.return_value = (
            '[["Zeroth Law", "Thermodynamics"],'
            ' ["First Law", "Energy"],'
            ' ["Second Law", "Entropy"]]'
        )

        await processor._generate_chunk_labels(
            sample_chunks
        )

        assert sample_chunks[0].chunk_labels == [
            "Zeroth Law", "Thermodynamics"
        ]
        assert sample_chunks[1].chunk_labels == [
            "First Law", "Energy"
        ]
        assert sample_chunks[2].chunk_labels == [
            "Second Law", "Entropy"
        ]

    async def test_empty_chunks(self, processor):
        """No-op for empty chunk list."""
        await processor._generate_chunk_labels([])
        processor.gemini.generate_text.assert_not_called()

    async def test_mismatch_falls_back(
        self, processor, sample_chunks
    ):
        """LLM returns fewer label lists than chunks."""
        processor.gemini.generate_text.return_value = (
            '[["Only One"]]'
        )

        await processor._generate_chunk_labels(
            sample_chunks
        )

        # All chunks should get heuristic labels
        for chunk in sample_chunks:
            assert chunk.chunk_labels is not None
            assert isinstance(chunk.chunk_labels, list)
            assert len(chunk.chunk_labels) > 0

    async def test_llm_failure_falls_back(
        self, processor, sample_chunks
    ):
        """LLM raises exception; heuristic applied."""
        processor.gemini.generate_text.side_effect = (
            RuntimeError("API error")
        )

        await processor._generate_chunk_labels(
            sample_chunks
        )

        for chunk in sample_chunks:
            assert chunk.chunk_labels is not None
            assert isinstance(chunk.chunk_labels, list)
            assert len(chunk.chunk_labels) > 0
            assert all(
                isinstance(l, str) and len(l) > 0
                for l in chunk.chunk_labels
            )


@pytest.mark.asyncio
class TestLabelSingleBatch:
    """Tests for _label_single_batch method."""

    async def test_batch_prompt_contains_excerpts(
        self, processor
    ):
        """Prompt includes truncated excerpts."""
        texts = ["Text one.", "Text two."]
        processor.gemini.generate_text.return_value = (
            '[["Label 1"], ["Label 2"]]'
        )

        result = await processor._label_single_batch(
            texts
        )

        call_args = (
            processor.gemini.generate_text.call_args
        )
        prompt = call_args[0][0]
        assert 'Excerpt 1:' in prompt
        assert 'Excerpt 2:' in prompt
        assert result == [["Label 1"], ["Label 2"]]

    async def test_batch_truncates_to_200_chars(
        self, processor
    ):
        """Long texts are truncated to 200 chars."""
        long_text = "A" * 500
        processor.gemini.generate_text.return_value = (
            '[["Long Label"]]'
        )

        await processor._label_single_batch([long_text])

        call_args = (
            processor.gemini.generate_text.call_args
        )
        prompt = call_args[0][0]
        # Excerpt should contain max 200 A's
        assert "A" * 201 not in prompt
